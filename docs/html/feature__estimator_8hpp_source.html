141  Eigen::Matrix<T, 3, 3> quatToRot(const Eigen::Matrix<T, 4, 1> &q) const {
142  const T q1 = q(0);
143  const T q2 = q(1);
144  const T q3 = q(2);
145  const T q4 = q(3);
146 
147  Eigen::Matrix<T, 3, 3> R;
148  R(0, 0) = 1.0 - 2.0 * pow(q2, 2.0) - 2.0 * pow(q3, 2.0);
149  R(0, 1) = 2.0 * (q1 * q2 + q3 * q4);
150  R(0, 2) = 2.0 * (q1 * q3 - q2 * q4);
151 
152  R(1, 0) = 2.0 * (q1 * q2 - q3 * q4);
153  R(1, 1) = 1.0 - 2.0 * pow(q1, 2.0) - 2.0 * pow(q3, 2.0);
154  R(1, 2) = 2.0 * (q2 * q3 + q1 * q4);
155 
156  R(2, 0) = 2.0 * (q1 * q3 + q2 * q4);
157  R(2, 1) = 2.0 * (q2 * q3 - q1 * q4);
158  R(2, 2) = 1.0 - 2.0 * pow(q1, 2.0) - 2.0 * pow(q2, 2.0);

177  Eigen::Matrix<T, 3, 3> K = this->formK<T>();
178 
179  // Form rotation matrix from quaternion q = (x, y, z, w)
180  const Eigen::Matrix<T, 4, 1> q{cam_q[0], cam_q[1], cam_q[2], cam_q[3]};
181  const Eigen::Matrix<T, 3, 3> R = this->quatToRot<T>(q);
182 
183  // Form landmark
184  const Eigen::Matrix<T, 3, 1> X{landmark[0], landmark[1], landmark[2]};
185 
186  // Form translation
187  const Eigen::Matrix<T, 3, 1> t{cam_p[0], cam_p[1], cam_p[2]};
188 
189  // Project 3D point to image plane
190  const Eigen::Matrix<T, 3, 1> est = K * R.transpose() * (X - t);
191 
192  // Convert projected point in homogenous coordinates to image coordinates
193  Eigen::Matrix<T, 2, 1> est_pixel;
194  est_pixel(0) = est(0) / est(2);
195  est_pixel(1) = est(1) / est(2);

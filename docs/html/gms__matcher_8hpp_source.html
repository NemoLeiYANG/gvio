99 inline cv::Mat draw_inliers(const cv::Mat &src1,

101  const std::vector<cv::KeyPoint> &kpt1,
102  const std::vector<cv::KeyPoint> &kpt2,
103  const std::vector<cv::DMatch> &inlier,
104  int type = 1) {
105  const int height = std::max(src1.rows, src2.rows);
106  const int width = src1.cols + src2.cols;
107  cv::Mat output(height, width, CV_8UC3, cv::Scalar(0, 0, 0));
108  src1.copyTo(output(cv::Rect(0, 0, src1.cols, src1.rows)));
109  src2.copyTo(output(cv::Rect(src1.cols, 0, src2.cols, src2.rows)));
110 
111  if (type == 1) {
112  for (size_t i = 0; i < inlier.size(); i++) {
113  cv::Point2f left = kpt1[inlier[i].queryIdx].pt;
114  cv::Point2f right =
115  (kpt2[inlier[i].trainIdx].pt + cv::Point2f((float) src1.cols, 0.f));
116  cv::line(output, left, right, cv::Scalar(0, 255, 255));
117  }
118  } else if (type == 2) {
119  for (size_t i = 0; i < inlier.size(); i++) {
120  cv::Point2f left = kpt1[inlier[i].queryIdx].pt;
121  cv::Point2f right =
122  (kpt2[inlier[i].trainIdx].pt + cv::Point2f((float) src1.cols, 0.f));
123  cv::line(output, left, right, cv::Scalar(255, 0, 0));
124  }
125 
126  for (size_t i = 0; i < inlier.size(); i++) {
127  cv::Point2f left = kpt1[inlier[i].queryIdx].pt;
128  cv::Point2f right =
129  (kpt2[inlier[i].trainIdx].pt + cv::Point2f((float) src1.cols, 0.f));
130  cv::circle(output, left, 1, cv::Scalar(0, 255, 255), 2);
